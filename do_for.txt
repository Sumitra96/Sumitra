#include <stdio.h>

int add();
int sub();
int mul();
int div();
int a;
int b;
int add(){

    
    printf("Enter numbers to add: \n");
    scanf("%d  %d" , &a ,&b);
    return (a+b);
}

int div(){

    printf("Enter numbers to div: \n");
    scanf("%d  %d" , &a ,&b);
    return (a/b);
}

int mul(){

    printf("Enter numbers to mul: \n");
    scanf("%d  %d" , &a ,&b);
    return (a*b);
}

int sub(){

    printf("Enter numbers to sub: \n");
    scanf("%d  %d" , &a ,&b);
    return (a-b);
}

int main(){

    int operation;
    char ca;
    int results[10];
    int operators[10];
    int i;
    char j;
    do{
    
    printf("Enter number to perform the operations:\n");
    scanf("%d" , &operation);
    
        switch(operation){
        case 1:
            
            results[i]=add();
            operators[j]=operation;
            break;
        
        case 2:

            results[i]=sub();
            operators[j]=operation;
            break;
        
        case 3:

            results[i]=div();
            operators[j]=operation;
            break;

        case 4:

            results[i]=mul();
            operators[j]=operation;
            break;
        
        default:
        printf("Invalid number");
        
        }
        
        printf("Would you like continue:(y/n)");
        scanf(" %c" , &ca);
        
    } while(ca=='y');
    
    for(int i=0; i<10;i++){
        
        printf("%d %c %d = %d" , a , operators[i] , b , results[j]);
    }
}
